001package pkg_commands;
002
003import pkg_EngineElements.*;
004/**
005 * Implementation of the 'back' user command.
006 * 
007 * @author Michael Kolling and David J. Barnes
008 * @version 2011.07.31
009 */
010public class BackCommand extends Command
011{
012    /**
013     * Constructor for objects of class BackCommand
014     */
015    public BackCommand()
016    {
017    }
018
019    /**
020     *  retourne le joueur en arrière 
021     *  verifie si il ne va pas dans une direction interdite
022     */
023    public void execute(GameEngine pGE)
024    {
025        if(getSecondWord() == null) {
026            if (! pGE.getPlayer().getHistory().empty()) {
027                Room vRoomPre = pGE.getPlayer().getHistory().pop();
028                if (vRoomPre.getDescription().equals("Entrée du manoir")){
029                    pGE.getUI().print("Vous ne pouvez pas revenir plus loin"); 
030                    pGE.getUI().println("car la porte s'est refermée quand vous êtes entré");
031                }
032                else{
033                    pGE.goToRoom(vRoomPre);
034                }
035            }
036            else pGE.getUI().println("Vous ne pouvez pas revenir plus loin");;
037        }
038        else {
039
040        }
041    }
042
043}
