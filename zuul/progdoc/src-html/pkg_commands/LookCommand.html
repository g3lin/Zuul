001package pkg_commands;
002import pkg_EngineElements.GameEngine;
003/**
004 * Implementation of the 'look' user command.
005 * 
006 * @author Michael Kolling and David J. Barnes
007 * @version 2011.07.31
008 */
009public class LookCommand extends Command
010{
011    /**
012     * Constructor for objects of class LookCommand
013     */
014    public LookCommand()
015    {
016    }
017
018    /**
019     *  regarde autour et affiche la long description de la rooom actuelle  
020     */
021    public void execute(GameEngine pGE)
022    {
023        if(getSecondWord() == null) {
024            pGE.getUI().println(pGE.getPlayer().getCurrentRoom().getLongDescription());
025        }
026        else {
027            pGE.getUI().println("Quoi ?");
028
029        }
030    }
031
032}
