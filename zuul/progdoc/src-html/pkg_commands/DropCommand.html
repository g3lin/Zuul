008/**
009 * Implementation of the 'drop' user command.
010 * 
011 * @author Michael Kolling and David J. Barnes
012 * @version 2011.07.31
013 */
014
015public class DropCommand extends Command
016{
017    /**
018     * Constructor for objects of class DropCommand
019     */
020    public DropCommand()
021    {
022    }
023
024    /**
025     *  prend un item dans l'inventaire du joueur et le mose dans la piece où le joueur est actuellement
026     *  @param pGE le GameEngine du jeu
027     */
028    public void execute(GameEngine pGE)
029    {
030        if(getSecondWord() == null) {
031            pGE.getUI().println("Quel objet voulez vous déposer ?");
032        }
033        else {
034            Item vItem = pGE.getPlayer().getItems().takeItem(getSecondWord());
035            if (vItem != null ){
036                pGE.getPlayer().getCurrentRoom().getItems().setItem(vItem);
037                pGE.getPlayer().setPoids( pGE.getPlayer().getPoids() - vItem.getPoids() );
038            }
039
040            else {
041                pGE.getUI().println("objet non trouvé");
042            }
043            pGE.getUI().setSprites();
044
045        }
046       
047
048    }
049}
