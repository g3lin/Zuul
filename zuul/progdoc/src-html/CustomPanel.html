012import javax.swing.JPanel;
013import java.awt.Graphics;
014import java.awt.Graphics2D;
015import java.awt.RenderingHints;
016import java.util.HashMap;
017import java.awt.*;
018import java.awt.event.MouseListener;
019import java.awt.event.MouseEvent;
020
021public class CustomPanel extends JPanel implements MouseListener {
022    private GameEngine aEngine;
023    private Image aBGImage;
024    private HashMap<String,Sprite> aSprites ;
025    private Sprite aPlayerSprite;
026    private int bgWidth;
027    private int bgHeight;
028
029    public CustomPanel(final GameEngine pGameEngine)
030    {
031        super();
032        this.aEngine = pGameEngine;
033        bgWidth =500;
034        bgHeight =800;
035        aSprites = new HashMap<String,Sprite>();
036        addMouseListener(this);
037    }
038
039    @Override
040    public void paintComponent(Graphics g) {
041        super.paintComponent(g);
042        Graphics2D g2d = (Graphics2D) g;
043        g2d.setRenderingHint(RenderingHints.KEY_ANTIALIASING,
044            RenderingHints.VALUE_ANTIALIAS_ON);
045        g2d.drawImage(scaleBG(this.aBGImage), 0, 0, this);
046        for (Sprite vSprite : this.aSprites.values()){
047            if(vSprite.estVisible()){
048                int adjustedHeight = vSprite.getHeight()*this.bgHeight/100;
049                vSprite.setAdjustedHeight(adjustedHeight);
050                Image rescaledImage = vSprite.getImage().getScaledInstance(-1,adjustedHeight,Image.SCALE_SMOOTH);;
051                vSprite.setAdjustedWidth(rescaledImage.getHeight(this));
052                int adjustedX = vSprite.getX()*this.bgWidth/100 ;
053                int adjustedY = vSprite.getY()*this.bgHeight/100 ;
054                g2d.drawImage(rescaledImage, adjustedX, adjustedY, this);
055            }
056        }
057    }
058
059    @Override
060    public Dimension getPreferredSize() {
061        if (this.getSize().equals(new Dimension(0,0))){
062            return new Dimension(800,1000);
063        }
064        else {
065            return new Dimension(this.bgHeight, this.bgWidth);
066        }
067    }
068
069    public Image scaleBG(final Image pImage){
070        Image vImg = pImage;
071        try{
072            vImg = pImage.getScaledInstance(getWidth(),-1,pImage.SCALE_SMOOTH);
073            this.bgHeight = vImg.getHeight(this);
074            this.bgWidth = vImg.getWidth(this);
075        }
076        catch(NullPointerException e){System.out.println("bug ");}
077        //requis pour eviter des exceptions au demarrage du prog 
078        //(dimensions pas encore initialisées au 1er passage
079
080        return vImg;
081    }
082
083    public void setBGImage(final Image pImage){
084        this.aBGImage = pImage;
085        repaint();
086    }
087
088    public void addSprite(final Sprite pSprite){
089        this.aSprites.put(pSprite.getName(),pSprite);
090        repaint();
091    }
092
093    public void setAsPlayerSprite( final Sprite pSprite){
094        this.aPlayerSprite = pSprite;
095        addSprite(pSprite);
096    }
097
098    public void resetSprites(){
099        this.aSprites = new HashMap<String,Sprite>();
100        repaint();
101    }

110
111    }

115
116    }

120

123    @Override
124    public void mouseClicked(MouseEvent pEvent) {
125        int vX = pEvent.getX()*100/this.bgWidth;
126        int vY = pEvent.getY()*100/this.bgHeight;
127
128        //CLIC GAUCHE
129        if (pEvent.getButton() == MouseEvent.BUTTON1) {
130            this.aPlayerSprite.tpTo(vX,this.aPlayerSprite.getY());
131
132            // CLICS POUR LES SORTIES
133            // else ifs pour la priorité du deplacement sur x par rapport à celui sur y
134            // cad si clic sur coin sup droit ---> dep à droite et par en haut
135            if (vX<15){
136                this.aEngine.interpretCommand("go west");
137                this.aPlayerSprite.tpTo(100-this.aPlayerSprite.getX()-20,this.aPlayerSprite.getY());
138            }
139            else if (vX>85){
140                this.aEngine.interpretCommand("go east");
141                this.aPlayerSprite.tpTo(100-this.aPlayerSprite.getX(),this.aPlayerSprite.getY());
142            }
143            else if (vY<15){
144                this.aEngine.interpretCommand("go north");
145                //                 this.aPlayerSprite.tpTo(this.aPlayerSprite.getX(),100-this.aPlayerSprite.getY());
146            }
147            else if (vY>85){
148                this.aEngine.interpretCommand("go south");
149                //                 this.aPlayerSprite.tpTo(this.aPlayerSprite.getX(),100-this.aPlayerSprite.getY()+20);
150            }
151
152            // CLICS SUR LES OBJETS
153            String vItemName = null;
154            for (Sprite vSprite : this.aSprites.values()){
155                boolean x1 = vX > vSprite.getX();
156                boolean x2 = vX < vSprite.getX()+ 15;
157                boolean y1 = vY > vSprite.getY();
158                boolean y2 = vX < vSprite.getX()+ vSprite.getHeight();
159                if(( x1 && x2)&&( y1 && y2)){
160                    vItemName = vSprite.getName();
161                }
162            }
163
164            if (vItemName == null){
165                this.aEngine.getUI().setButtons(new String[]{"look","eat","back"});
166            }
167
168            else if (vItemName.equals("player")){
169                String[] vSArray;
170                if (! this.aEngine.getPlayer().getItems().isEmpty() ) {
171                    Item[] vIArray = this.aEngine.getPlayer().getItems().getItemArray();
172                    vSArray = new String[vIArray.length+1];
173                    int i = 0;
174                    for (Item vI : vIArray){
175                        vSArray[i]= "drop "+vI.getName();
176                    }
177                     vSArray[vIArray.length] = "inventaire";
178                }
179                else {
180                    vSArray = new String[]{"inventaire"};
181                }
182                this.aEngine.getUI().setButtons(vSArray);
183            }
184            
185            else{
186                this.aEngine.getUI().setButtons(new String[]{"take "+vItemName,"use "+vItemName});
187            }
188        }
189
190        //CLIC MILIEU
191        if (pEvent.getButton() == MouseEvent.BUTTON2) {
192            this.aPlayerSprite.tpTo(vX,vY);
193        }
194
195        //CLIC DROIT
196        if (pEvent.getButton() == MouseEvent.BUTTON3) {
197            this.aPlayerSprite.tpTo(50,50);
198            System.out.println(pEvent.getX()+"  , bg width "+this.bgWidth);
199        }
200
201    }
