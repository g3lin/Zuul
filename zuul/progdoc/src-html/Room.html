001import java.util.Set;
002import java.util.HashMap;
003import java.util.Map;
004import java.util.Iterator;
005
006public class Room
007{
008    private String aDescription ;
009    private HashMap<String,Room> aExits;
010    private String aImageName;
011    private ItemList aItems;
012
013    /**
014     * Constructeur mettant en place la description
015     * et la hashmap des sorties en attributs
016     * @param pDescription la description de la piece
017     * @param pImage le nom de l'image
018     */
019    public Room(final String pDescription, String pImage){
020        this.aDescription=pDescription;
021        this.aExits = new  HashMap<String,Room>();
022        this.aImageName = pImage;
023        this.aItems= new ItemList();
024    }
025
026    /**
027     * accesseur de la description
028     * @return la description
029     */
030    public String getDescription() {
031        return this.aDescription;
032    }
033
034    /**
035     * definit les sorties
036     * @param pDirection la direction
037     * @param pExit  la sortie pour la direction
038     */
039    public void setExits(String pDirection, Room pExit){
040        aExits.put(pDirection,pExit);
041
042    }
043
044    /**
045     * accesseur de la direction
046     * @param pDirection la direction
047     * @return la room qui fait face à cette direction
048     */
049    public Room getExit(final String pDirection){
050        return aExits.get(pDirection);
051    }
052
053    /**
054     * affiche le string de la localisation
055     */
056    public String getLongDescription(){
057        String vDescription = "Vous êtes dans "+ this.getDescription()+"\n Vous pouvez aller vers: "+this.getExitString() ;
058        if (! this.aItems.isEmpty() ) {
059            vDescription += "\n Vous trouvez dans cette salle :"+this.aItems.inventory();
060        }
061        return vDescription;
062    }
063
064    /**
065     * affiche le string des sorties disponibles en sortant de cette salle
066     */
067    public String getExitString(){
068        StringBuilder vExits = new StringBuilder( "Exits:" );
069        for ( String vS : aExits.keySet() )
070            vExits.append( " " + vS );
071        return vExits.toString();
072    }
073
074    public ItemList getItems(){
075        return this.aItems;
076    }
077
078    /**
079     * Return a string describing the room's image name
080     */
081    public String getImageName()
082    {
083        return aImageName;
084    }
085} // Room
