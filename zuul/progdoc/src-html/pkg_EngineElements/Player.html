013public class Player

016    private Room aCurrentRoom;

023    /**
024     * Constructor for objects of class Player
025     */
026    public Player()
027    {
028        this.aHistory = new Stack<Room>();
029        this.aCurrentRoom = null;
030        this.aItems= new ItemList();
031        this.aPoids = 0;
032        this.aPoidsMax = 1000;
033        this.aSprite = new Sprite( "player", "Images/character.png" , 50,50, 33);

036    }

038    /**

040     * @param pRoom la salle
041     */
042    public void setCurrentRoom(final Room pRoom)
043    {
044        this.aCurrentRoom = pRoom;
045    }
046
047    /**
048     * retourne le sprite du joueur
049     * @return sprite du joueur
050     */
051    public Sprite getSprite(){
052        return this.aSprite;
053    }
054
055    /**
056     * donne la room dans laquelle le joueur est
057     * @return room actuelle
058     */
059    public Room getCurrentRoom(){
060        return this.aCurrentRoom;
061    }
062
063    /**
064     * donne la l'historiqque de deplacement du joueur
065     * @return stack des deplacements précédents
066     */
067    public Stack<Room> getHistory(){

069    }

090    public int getPoids(){

092    }
093
094    /**
095     * donne le poids max du joueur
096     * @return le poids  max
097     */
098    public int getPoidsMax(){
099        return this.aPoidsMax;
100    }
101
102    /**
103     * met le poids du joueur
104     * @param pPoids le poids total de l'inventaire
105     */
106    public void setPoids(final int pPoids){
107        this.aPoids = pPoids;
108    }
109
110    /**
111     * met le poids max  du joueur

114    public void setPoidsMax(final int pPoids){

117
118    /**
119     * affiche tous les objets du joueur
120     * @return String contenant tous les objets que le joueur porte
121     */ 
122    
123    public String getInventory(){
124        String vString = "";
125        if (! this.aItems.isEmpty() ) {
126            vString +=  "Votre inventaire contient :"+this.aItems.inventory();
127        }
128        else {
129            vString += "Votre inventaire est vide";
130        }
131        vString += "\n Le poids de votre equipement est de : "+getPoids()+"g";
132        return vString;
133    }
