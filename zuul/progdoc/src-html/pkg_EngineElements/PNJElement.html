009public class PNJElement extends Item

011    // instance variables - replace the example below with your own
012    private String[] aDialogues;
013    private int aDialInt;
014    private Room[] aDepPattern;
015
016    private GameEngine aGE;
017    private Room aRoom;
018
019    /**
020     * Constructor for objects of class Character
021     */
022    public PNJElement(final String pName, final String pDescription, final String[] pDialogues, final Room[] pDepPattern, final GameEngine pGE)
023    {
024        super(9999,pName, pDescription);
025        this.aDialogues = pDialogues;
026        this.aDepPattern = pDepPattern;
027        this.aGE = pGE;
028        this.aDialInt =0;
029    }
030
031    /**
032     *  fait parler le PNJ et verifie si il doit se deplacer
033     */
034    public String talk()
035    {
036        if (this.aDialInt <= this.aDialogues.length)
037        {
038            this.aDialInt ++ ;
039        }
040
041        if (this.aDepPattern[this.aDialInt-1] != null){
042            Room vR = this.aDepPattern[this.aDialInt-1];
043            this.aGE.movePNJRoom(this,vR);
044        }
045
046        return this.aDialogues[this.aDialInt -1];
047    }
048
049    /**
050     * definit la room où est le pnj
051     * @param pRoom la salle 
052     */
053    public void setRoom(final Room pRoom){
054        this.aRoom = pRoom;
055    }
056
057    
058    /**
059     * donne la room où est le pnj
060     * @return la salle
061     */
062    public Room getRoom(){
063        return this.aRoom;
064    }
