001package pkg_EngineElements;
002
003import java.util.Random;
004/**
005 * Salle speciale qui tp dans une salle au hasard
006 * 
007 * @author GELIN
008 * @version 2017
009 */
010public class TransporterRoom extends Room
011{
012    private Room aPseudoRandom;
013    private Room[] aAvailableRooms;
014    /**
015     * Constructor for objects of class TransporterRoom
016     */
017    public TransporterRoom (final String pDesc, final String pImg, final Room[] pARooms)
018    {
019        super(pDesc,pImg);
020        this.aPseudoRandom = null;
021        this.aAvailableRooms = pARooms;
022    }
023
024    @Override
025    /**
026     * accesseur de la direction
027     * @param pDirection la direction
028     * @return la room au hasard ou souhaitée si on afait alea
029     */
030    public Room getExit(final String pDirection){
031        if (pDirection.equals("west")){
032            return findARandomRoom();
033        }
034        else{            
035            return null;
036        }
037    }
038
039    /**
040     * retoune la string des sorties modifiée pour la salle
041     */
042    @Override
043    public String getExitString(){
044        return "une salle au hasard dans la maison à portée de l'appareil";
045    }
046
047    /**
048     * trouve une salle random
049     * @return salle random
050     */
051    public Room findARandomRoom(){
052        if (aPseudoRandom == null){
053            Random vRandomObject = new Random();
054            int vI = vRandomObject.nextInt(this.aAvailableRooms.length); //DERNIER NON INCLUS
055            return aAvailableRooms[vI];
056        }
057        else{
058            return this.aPseudoRandom;
059        }
060    }
061
062    /**
063     * Truque le jeu et met une salle définie au lieu de random
064     * @param pRoom la salle vers laquelle il faut aller
065     */
066    public void setPseudoRandom(final Room pR){
067        this.aPseudoRandom = pR;
068    }
069
070}
