036/**
037 * This class implements a simple graphical user interface with a text entry
038 * area, a text output area and an optional image.
039 * 
040 * @author Michael Kolling (DB edited)
041 * @version 1.0 (Jan 2003)
042 */
043public class UserInterface implements ActionListener
044{
045    private GameEngine aEngine;
046    private JFrame     aMyFrame;
047    private JTextField aEntryField;
048    private JTextPane  aLog;
049    private JLabel  aTimer;
050    private CustomPanel aUI;
051    private Stack<JButton> aStackButtons;
052    private JPanel aButtonsHolder;
053    /**
054     * Construct a UserInterface. As a parameter, a Game Engine
055     * (an object processing and executing the game commands) is
056     * needed.
057     * 
058     * @param gameEngine  The GameEngine object implementing the game logic.
059     */
060    public UserInterface( final GameEngine pGameEngine )
061    {
062        this.aEngine = pGameEngine;
063        this.createGUI();
064    } // UserInterface(.)
065
066    /**
067     * Print out some text into the text area.
068     */
069    public void print( final String pText )
070    {
071        try{
072            this.aLog.getDocument().insertString(this.aLog.getDocument().getLength(),pText, null );
073        }
074        catch(javax.swing.text.BadLocationException exc){System.out.print("bug du texte");}
075        this.aLog.setCaretPosition( this.aLog.getDocument().getLength() );
076    } // print(.)
077
078    /**
079     * Print out some text into the text area, followed by a line break.
080     */
081    public void println( final String pText )
082    {
083        this.print( pText + "\n" );
084    } // println(.)
085
086    /**
087     * Show an image file in the interface.
088     */
089    public void showImage( final String pImageName )
090    {
091        Image vImage = null;
092        try {
093            vImage = ImageIO.read(new File(pImageName));
094        } catch (IOException e) {
095
096        }
097        this.aUI.setBGImage(vImage);
098    } // showImage(.)
099
100    /**
101     * Enable or disable input in the input field.
102     */
103    public void enable( final boolean pOnOff )
104    {
105        this.aEntryField.setEditable( pOnOff );
106        if ( ! pOnOff )
107            this.aEntryField.getCaret().setBlinkRate( 0 );
108    } // enable(.)
109
110    /**
111     * Set up graphical user interface.
112     */
113    private void createGUI()
114    {
115        this.aMyFrame = new JFrame( "Le Manoir de WanderStock" );
116        ImageIcon img = new ImageIcon("Images/icon.png");
117        this.aMyFrame.setIconImage(img.getImage());
118        this.aEntryField = new JTextField( 34 );
119        this.aEntryField.setMaximumSize(new Dimension(1000,50));
120
121        this.aLog = new JTextPane();
122        this.aLog.setEditable( false );
123        JScrollPane vListScroller = new JScrollPane( this.aLog );
124        vListScroller.setPreferredSize( new Dimension(200, 200) );
125        vListScroller.setMinimumSize( new Dimension(100,50) );
126
127        JPanel vPanel = new JPanel();
128        vPanel.setLayout( new BoxLayout(vPanel, BoxLayout.PAGE_AXIS) );
129
130        // PLACEMENT DE L'UI PRINCIPALE
131        this.aUI = new CustomPanel(this.aEngine);
132        this.aUI.setPreferredSize( new Dimension(100,100) );
133        vPanel.add( this.aUI);
134
135        // PLACEMENT DES BOUTONS
136        this.aStackButtons = new Stack<JButton>();
137        this.aButtonsHolder = new JPanel();
138        JButton vButton1 = new JButton("eat");
139        JButton vButton2 = new JButton("look");
140        JButton vButton3 = new JButton("back");
141
142        addButton(vButton1);
143        addButton(vButton2);
144        addButton(vButton3);
145
146        this.aTimer = new JLabel("timer");
147        vPanel.add(this.aTimer);
148        vPanel.add(this.aButtonsHolder);
149
150        //PLACEMENT DE LA ZONE DE TEXTE
151        try{
152            Font vFont = Font.createFont(Font.TRUETYPE_FONT, new File("font.ttf"));
153            GraphicsEnvironment.getLocalGraphicsEnvironment().registerFont(vFont);
154            Font vPS2PFont = new Font("Press Start 2P", Font.PLAIN, 15);
155            this.aLog.setFont(vPS2PFont);
156        }
157        catch(Exception e){}
158
159        vPanel.add( vListScroller);
160
161        // PLACEMENT DE LA ZONE D'ENTRÉE TEXTE
162        vPanel.add( this.aEntryField);
163
164        // INITIALISATION DE LA FENETRE
165        this.aMyFrame.getContentPane().add( vPanel);
166
167        // add some event listeners to some components
168
169        this.aMyFrame.addWindowListener( new WindowAdapter() {
170                public void windowClosing(WindowEvent e) { System.exit(0); }
171            } );
172
173        this.aEntryField.addActionListener( this );
174
175        this.aMyFrame.pack();
176        this.aMyFrame.setVisible( true );
177        this.aMyFrame.setSize( new Dimension(800,700) );
178        this.aUI.revalidate(); //requis pour eviter des bugs d'affichage
179        this.aEntryField.requestFocus();
180        setSprites();
181    } // createGUI()
182
183    public void resetButtons(){
184        this.aButtonsHolder.removeAll();
185        this.aStackButtons = new Stack<JButton>();
186    }
187
188    public void addButton( final JButton pB){
189        pB.addActionListener(this);
190        this.aButtonsHolder.add(pB);
191        this.aStackButtons.push(pB);
192    }
193
194    public void setButtons(final String[] pSArray){
195        resetButtons();
196        for (String vS : pSArray){
197            addButton(new JButton(vS));
198        }
199    }
200
201    public void setSprites(){
202        this.aUI.resetSprites();
203        Room vRoom = this.aEngine.getPlayer().getCurrentRoom();
204        for (Item vI :vRoom.getItems().getItemArray()){
205            Sprite vS = vI.getSprite();
206            this.aUI.addSprite(vS);
207        }
208        this.aUI.setAsPlayerSprite(this.aEngine.getPlayer().getSprite());
209
210    }
211
212    /**
213     * Actionlistener interface for entry textfield.
214     */
215    public void actionPerformed( final ActionEvent pEvent ) 
216    {
217
218        if (pEvent.getSource() == aEntryField){
219            this.processCommand();
220        }
221
222        for (JButton vB : this.aStackButtons){
223            if (pEvent.getSource() == vB){
224                String vCommand = pEvent.getActionCommand();
225                this.aEngine.interpretCommand( vCommand );
226            }
227        }
228
229    } // actionPerformed(.)
230
231    /**
232     * A command has been entered. Read the command and do whatever is 
233     * necessary to process it.
234     */
235    private void processCommand()
236    {
237        String vInput = this.aEntryField.getText();
238        this.aEntryField.setText( "" );
239
240        this.aEngine.interpretCommand( vInput );
241    } // processCommand()
242
243    public void updateTimeGUI(final int pTime){
244        this.aTimer.setText(pTime/60+":"+pTime%60);
245
246    }
247} // UserInterface
