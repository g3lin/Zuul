119    /**
120     * procedure affichant via S.o.p l'intro du jeu ainsi que la localisation du joueur
121     */

123        aGUI.println("Welcome to the World of Zuul!");
124        aGUI.println("World of Zuul is a new, incredibly boring adventure game.");
125        aGUI.println(this.aCurrentRoom.getLongDescription());
126        aGUI.showImage(aCurrentRoom.getImageName());

129    /**
130     * methode permettant de prendre en charge les commandes
131     * en fonction du 1er mot de la commande, va voir les différentes actions à effectuer
132     * @param une command
133     * @return le fait ou non que le joueur veuille quitter
134     */
135    public void interpretCommand(final String pCommand){
136        aGUI.println(pCommand);
137        Command vCommand = aParser.getCommand(pCommand);
138        String vCommandWord = vCommand.getCommandWord();
139
140        if ( vCommandWord.equals("quit") ){
141            if(vCommand.hasSecondWord())
142                aGUI.println("Quit what?");
143            else quit();
144        }
145
146        else if (vCommandWord.equals("help")){
147            printHelp();
148
149        }
150
151        else if (vCommandWord.equals("go")){
152            goRoom(vCommand);
153
154        }
155
156        else if (vCommand.isUnknown() ) {
157            aGUI.println("I don't know what you mean...");
158
159        }
160
161        else if (vCommandWord.equals("look")){
162            look();
163
164        }
165
166        else if (vCommandWord.equals("eat")){
167            eat();
168
169        }
170
171        else if (vCommandWord.equals("back")){
172            if(vCommand.hasSecondWord())
173                aGUI.println("Je n'ai n'ai pas compris");
174            else back();
175        }
176
177        else {
178            aGUI.println("I don't know what you mean...");
179
180        }
181    }
182
183    /**
184     * procedure permettant de changer de pièce dans le jeu
185     * @param pCommand une command
186     */
187    public void goRoom(final Command pCommand) {
188
189        if (!pCommand.hasSecondWord()) {
190            aGUI.println("go where ?");
191            return;
192        }
193        String vDirection = pCommand.getSecondWord() ;
194        Room vNextRoom = this.aCurrentRoom.getExit(vDirection);
195        if (vNextRoom == null){
196            aGUI.println("There is no door !");
197            return ;
198        }
199        else{
200            this.aRoomS.push(this.aCurrentRoom);
201
202            goToRoom(vNextRoom);
203        }
204
205    }
206
207    public void goToRoom(final Room pRoom){
208        this.aCurrentRoom=pRoom;
209
210        this.aGUI.println(this.aCurrentRoom.getLongDescription());
211
212        if(this.aCurrentRoom.getImageName() != null)
213        {this.aGUI.showImage(this.aCurrentRoom.getImageName());}
214    }

244        aGUI.println( this.aParser.showCommands() );
245    }
246
247    public void back(){
248        if (! aRoomS.empty()) {
249            Room vRoomPre = this.aRoomS.pop();
250            goToRoom(vRoomPre);
251        }
252        else this.aGUI.println("Vous ne pouvez pas revenir plus loin");
253    }
