003import java.util.Timer;
004import java.util.TimerTask;
005
006public class TimerElement extends TimerTask
007{
008    private int aTime;
009    private GameEngine aGE;
010    private Timer aTimer;
011    
012    public TimerElement(){
013        this.aTime = 150;
014        
015    }
016    
017    public void pass(final GameEngine pE , final Timer pT){
018        this.aGE = pE;
019        this.aTimer =pT;
020        
021    }
022
023    @Override
024    public void run() {
025        this.aTime -= 1;
026        
027        this.aGE.updateTime(this.aTime);
028        
029        
030        if (this.aTime == 0){
031            this.aTimer.cancel();
032            this.aGE.gameOver();
033        }
034    }
035}
